95  inline std::vector<std::string> Wrap(const std::string &in, const std::string::size_type width, std::string::size_type firstlinewidth = 0)
96  {
97  // Preserve existing line breaks
98  const auto newlineloc = in.find('\n');
99  if (newlineloc != in.npos)
100  {
101  auto first = Wrap(std::string(in, 0, newlineloc), width);
102  auto second = Wrap(std::string(in, newlineloc + 1), width);
103  first.insert(
104  std::end(first),
105  std::make_move_iterator(std::begin(second)),
106  std::make_move_iterator(std::end(second)));
107  return first;
108  }
109  if (firstlinewidth == 0)
110  {
111  firstlinewidth = width;
112  }
113  auto currentwidth = firstlinewidth;
114 
115  std::istringstream stream(in);
116  std::vector<std::string> output;
117  std::string line;
118  bool empty = true;
119 
120  for (char c : in)
121  {
122  if (!isspace(c))
123  {
124  break;
125  }
126  line += c;
127  }
128 
129  while (stream)
130  {
131  std::string item;
132  stream >> item;
133  auto itemsize = Glyphs(item);
134  if ((line.length() + 1 + itemsize) > currentwidth)
135  {
136  if (!empty)
137  {
138  output.push_back(line);
139  line.clear();
140  empty = true;
141  currentwidth = width;
142  }
143  }
144  if (itemsize > 0)
145  {
146  if (!empty)
147  {
148  line += ' ';
149  }
150 
151  line += item;
152  empty = false;
153  }
154  }
155 
156  if (!empty)
157  {
158  output.push_back(line);
159  }
160  return output;
161  }
